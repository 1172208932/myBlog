    leader: '黄小杨',
    zhuawa: ['部部', '小可']
  };
  Object.freeze(obj2); // freeze无法冻结嵌套引用类型

  // *如何破局
  // freeze如何做deep化。
  // 思路： 嵌套遍历并且逐层freeze
  function deepFreeze(obj = {}) {
    Object.freeze(obj);
    (Object.key(obj) || []).forEach(key => {
      if(type obj[key] === 'object') {
        deepFreeze(obj[key]);
      }
    })
  }

# 继承 - js如何继承

  // ES5 继承
  function Course(teacher, course) {
    this._teacher = teacher;
    this.course = course;
  }
  Course.call = function() {
    console.log('calling');
  }
  Course.prototype.send = function() {
    console.log('sending');
  }

  // 如何继承
  // 子对象
  function Child() {
    // 初始化父类
    Course.call(this, '云隐', 'ES6');
    this.start = function() {
      console.log('starting');
    }
  }
  Child.prototype = Course.prototype;

  // ES6
  // 父类
  class Course {
    constructor(teacher, course) {
      this._teacher = teacher;
      this.course = course;
    }
    send() {
      console.log('sending');
    }
    static call() {
      console.log('calling');
    }
  }

  // 子类
  class Child extends Course {
    constructor() {
      super('云隐', 'ES6');

    }
  }
