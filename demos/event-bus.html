    # Event-bus

    简单的发布订阅通信类（类似eventemitter3 (opens new window)）

    # 📦 Install

    yarn add @duiba/event-bus

    # Usage

    import EventBus from '@duiba/event-bus';

    # 技术栈：

    rollup/ts/...

    # 🔨 示例

    import EventBus from '@duiba/event-bus';
export default class App extends React.Component {
    constructor (props) {
        super(props)
        this.state = { message: 'Hello World' }
    }
    componentDidMount() {
        EventBus.on('BEGIN', this.immediatelyBegin, this);
    }
    componentWillUnmount() {
        EventBus.off('BEGIN', this.immediatelyBegin);
    }
    render () {
        return (
        <div className="box-react">
            React {this.state.message}
        </div>
        )
    }
}
